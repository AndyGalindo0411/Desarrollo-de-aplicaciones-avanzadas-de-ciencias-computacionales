programa factorial_rec;
vars n, res: entero;

nula factorial_rec(n: entero) {
  {
    si (n == 0) {
      res = 1;
    } sino {
      factorial_rec(n - 1);
      res = res * n;
    };
  }
};

inicio {
  escribe("--- FACTORIAL RECURSIVO ---");
  n = 5;
  factorial_rec(n);
  escribe("n =", n);
  escribe("Factorial recursivo:", res);
} fin
